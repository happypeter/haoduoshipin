~~~
dpkg-deb --build  debian/
sudo apt-get install dpkg-dev
dpkg-name debian.deb
sudo dpkg -i happyhello_1.0_i386.deb
sudo apt-get install hello
~~~

~~~
sudo apt-get remove happyhello
~~~
